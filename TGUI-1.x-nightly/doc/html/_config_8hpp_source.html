145// Enable the use of std::filesystem if TGUI is built with c++17 with a new enough compiler.
146// Although GCC and clang supported it before version 9, this is the first version where no
147// additional library has to be linked in order to use std::filesystem. This is also the
148// reason why we can't rely on __cpp_lib_filesystem for this.
149// Xcode 11 supports it on macOS 10.15 only, so we require Xcode 12 which doesn't run on macOS 10.14.
150#if TGUI_COMPILED_WITH_CPP_VER >= 17
151 #if (defined(__apple_build_version__) && defined(__clang_major__) && (__clang_major__ >= 12)) \
152 || (!defined(__apple_build_version__) && defined(__clang_major__) && (__clang_major__ >= 9)) \
153 || (defined(__GNUC__) && (__GNUC__ >= 9)) \
154 || (defined(_MSC_VER) && (_MSC_VER >= 1914))
155 #define TGUI_USE_STD_FILESYSTEM
