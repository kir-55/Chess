220 template <typename Func, typename... BoundArgs>
221 unsigned int SignalManager::connect(String widgetName, String signalName, Func&& handler, const BoundArgs&... args)
222 {
223 const unsigned int id = generateUniqueId();
224
225 if constexpr (std::is_convertible_v<Func, std::function<void(const BoundArgs&...)>>
226 && std::is_invocable_v<decltype(&handler), BoundArgs...>
227 && !std::is_function_v<Func>)
228 {
229 // Reference to function
230 m_signals[id] = {widgetName, signalName, makeSignal([=, f=std::function<void(const BoundArgs&...)>(handler)]{ std::invoke(f, args...); })};
231 }
232 else if constexpr (std::is_convertible_v<Func, std::function<void(const BoundArgs&...)>>)
233 {
234 // Function pointer
235 m_signals[id] = {widgetName, signalName, makeSignal([=]{ std::invoke(handler, args...); })};
236 }
237 else if constexpr (std::is_convertible_v<Func, std::function<void(const BoundArgs&..., const std::shared_ptr<Widget>&, const String&)>>
238 && std::is_invocable_v<decltype(&handler), BoundArgs..., const std::shared_ptr<Widget>&, const String&>
239 && !std::is_function_v<Func>)
240 {
241 // Reference to function with caller arguments
242 m_signals[id] = {widgetName, signalName, makeSignalEx([=, f=std::function<void(const BoundArgs&..., const std::shared_ptr<Widget>& w, const String& s)>(handler)](const std::shared_ptr<Widget>& w, const String& s){ std::invoke(f, args..., w, s); })};
243 }
244 else
245 {
246 // Function pointer with caller arguments
247 m_signals[id] = {widgetName, signalName, makeSignalEx([=](const std::shared_ptr<Widget>& w, const String& s){ std::invoke(handler, args..., w, s); })};
248 }
249
250 connect(id);
251 return id;
252 }
